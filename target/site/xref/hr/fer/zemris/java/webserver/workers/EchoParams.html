3   import java.io.IOException;
4   
5   import hr.fer.zemris.java.webserver.IWebWorker;
6   import hr.fer.zemris.java.webserver.RequestContext;
7   /**
8    * Implementation of {@link IWebWorker} interface. It prints out 
9    * table of parameter name and value pairs given in client request.
10   * 
11   * @author Vilim StaroveÅ¡ki
12   *
13   */
14  public class EchoParams implements IWebWorker {
15  
16  	@Override
17  	public void processRequest(RequestContext context) {
18  		
19  		context.setMimeType("text/html");
20  		try {
21  			context.write("<html>\r\n  <body>\r\n");
22  			context.write("    <h1>Tablica</h1>\r\n");
23  			context.write("      <table border='1'>\r\n");
24  			for(String name : context.getParameterNames()) {
25  				context.write("      <tr><td>");
26  				context.write(name);
27  				context.write("</td><td>");
28  				context.write(context.getParameter(name));
29  				context.write("</td></tr>\r\n");
30  			}
31  			context.write("      </table>\r\n  </body>\r\n</html>");
32  		} catch (IOException ex) {
33  			System.err.println("There was an IO error occured while writing to client.");
34  			System.err.println(ex.getMessage());
35  		}
36  		try {
37  			context.close();
38  		} catch (IOException e) {
39  			e.printStackTrace();
40  		}
41  	}
42  
43  }
