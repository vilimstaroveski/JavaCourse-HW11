4   /**
5    * Class that stores a value that represents a number. Supported types are Integer, Double,
6    * String(but number parsable) and null(parsed to Integer with value 0). 
7    * 
8    * @author Vilim StaroveÅ¡ki
9    *
10   */
11  public class ValueWrapper {
12  	
13  	/**
14  	 * Stored value.
15  	 */
16  	private Object value;
17  	/**
18  	 * Creates new {@link ValueWrapper}
19  	 * @param value that is going to be put inside
20  	 */
21  	public ValueWrapper(Object value) {
22  		
23  		this.value = value;
24  	}
25  	
26  	/**
27  	 * Adds stored value with value given as parameter.
28  	 * 
29  	 * @param incValue value that is added to stored value.
30  	 */
31  	public void increment(Object incValue) {
32  		
33  		NumberOperation operator = new NumberOperation(this.value, incValue) {
34  			
35  			@Override
36  			public Number doAnOperation() {
37  				TypeEnum typeOfResult = getFinalType();
38  				if(typeOfResult == TypeEnum.TYPE_INTEGER) {
39  					return getO1().intValue() + getO2().intValue();
40  				}
41  				else {
42  					return getO1().doubleValue() + getO2().doubleValue();
43  				}
44  			}
45  		};
46  		this.value = operator.doAnOperation();
47  	}
48  	
49  	/**
50  	 * Subtracts stored value with value given as parameter.
51  	 * 
52  	 * @param decValue value that is being subtracted from stored value.
53  	 */

55  
56  		NumberOperation operator = new NumberOperation(this.value, decValue) {
57  	
58  			@Override
59  			public Number doAnOperation() {

61  				if(typeOfResult == TypeEnum.TYPE_INTEGER) {

63  				}
64  				else {
65  					return getO1().doubleValue() - getO2().doubleValue();

67  			}

118 	/**
119 	 * Compares stored value to a value given as parameter. If stored value is greater, method will return 1
120 	 * if the values are equal, method returns 0 and if value is lesser, method will return -1.
121 	 * @param withValue value that the stored value is being compared to.
122 	 * @return 1 if stored value is greater, -1 if the stored value is lesser and 0 if the stored value is equal.
123 	 */
124 	public int numCompare(Object withValue) {
125 		
126 		NumberOperation operator = new NumberOperation(this.value, withValue) {
127 			
128 			@Override
129 			public Number doAnOperation() {
130 				return Double.compare(getO1().doubleValue(), getO2().doubleValue());
131 			}
132 		};
133 		return operator.doAnOperation().intValue();
134 	}
